安装依赖软件

sudo apt install ninja-build pixman

sudo apt-get install build-essential libcairo2-dev libpango1.0-dev libjpeg-dev libgif-dev librsvg2-dev

apt-get install libmount-dev

sudo apt install android-sdk-ext4-utils

sudo apt install scons

sudo apt install device-tree-compiler

clone仓库

git clone https://github.com/RT-Thread/rtthread-pomegranate.git

下载源码

./dn.sh
// 如果下载不成功手动下载到`tarball`目录

下载工具链

// 下载后解压到源码目录下的`toolchains`目录
https://toolchains.bootlin.com/downloads/releases/toolchains/riscv64/tarballs/riscv64--glibc--bleeding-edge-2020.08-1.tar.bz2

编译u-boot

cd u-boot
./build.sh

生成根文件系统

cd buildroot
./build.sh

编译linux

cd linux
./build.sh

修改linux下run.sh脚本

${QEMU_RUNTIME} -nographic -smp 2 -machine virt -m 512M -kernel linux-5.12.3/arch/riscv/boot/Image -append "root=/dev/vdarw console=ttyS1,115200n8 earlycon=sbi" -drive file=${ROOT}/output/rootfs.ext4,format=raw,id=hd0 -device virtio-blk-device,drive=hd0

编译opensbi

cd opensbi
./build.sh

下载工具链

cd toolchains
// https://github.com/sifive/freedom-tools/releases
wget https://static.dev.sifive.com/dev-tools/freedom-tools/v2020.12/riscv64-unknown-elf-toolchain-10.2.0-2020.12.8-x86_64-linux-ubuntu14.tar.gz
tar -xf riscv64-unknown-elf-toolchain-10.2.0-2020.12.8-x86_64-linux-ubuntu14.tar.gz

// 添加工具链到环境变量，例如：
export PATH=/home/ubuntu/rtthread-pomegranate/toolchains/riscv64-unknown-elf/bin:"$PATH"

编译rtt

// clone仓库
cd rtos
mkdir rt-thread
git clone https://github.com/RT-Thread/rt-thread.git

// 切换到最新代码，当前是v4.1.0-beta：
git checkout v4.1.0-beta

./build.sh

// 运行
./run.sh
